# Heirs party formation

character_event = {
	id = party.1
	desc = EVTDESCPARTY1
	picture = GFX_evt_emissary
	
	trigger = {
		AND = {
			age = 16
			top_liege = { primary_title = { has_law = vice_royalty_2 } }
			independent = no
			NOT = {
				trait = heirs_party
				trait = first_generation_party
			}
			OR = {
				is_councillor = yes
				#has_minor_title =   #Add minor titles e.g. Agriculture Minister
				#has_minor_title = royal_army_officer
				is_ruler = yes
				#trait = [FORMER BUREAUCRAT] #This exhausts the list of bureaucratic positions
			}
			OR = {
				father_even_if_dead = {
					OR = {
						is_councillor = yes
						#has_minor_title =   #Add minor titles e.g. Agriculture Minister
						#has_minor_title = royal_army_officer
						is_ruler = yes
						#trait = [FORMER BUREAUCRAT] #This fails to check whom said forebearer served, might be foreign bureaucrat
					}
				}
				mother_even_if_dead = {
					OR = {
						is_councillor = yes
						#has_minor_title =   #Add minor titles e.g. Agriculture Minister
						#has_minor_title = royal_army_officer
						is_ruler = yes
						#trait = [FORMER BUREAUCRAT] #This fails to check whom said forebearer served, might be foreign bureaucrat
					}
				}
			}
		}
	}
	
	mean_time_to_happen = {
		years = 5 #This might be changed to manipulate party density
		modifier = {
			factor = 0.5
			trait = hereditary_party
		}
	}
	
	option = {
		name = EVTOPTAPARTY1 #Party created
		add_trait = heirs_party
		factor = 100
	}
	
	option = {
		name = EVTOPTBPARTY1 #Party not created
		ai_chance = {
			factor = 75
			modifier = {
				factor = 2
				trait = meritocratic_party
			}
			modifier = {
				factor = 0.1
				trait = hereditary_party
			}
		}
	}
}
